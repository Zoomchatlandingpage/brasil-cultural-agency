### **2. NOVA ROTA DE CRIAÇÃO DE CONTA**
```typescript
// server/routes.ts
app.post('/api/client/create-account', async (req, res) => {
  const { email, packageData } = req.body;
  
  // Gerar credenciais automáticas
  const username = `BrazilExplorer${Math.floor(Math.random() * 9999)}`;
  const password = generatePassword();
  
  // Criar usuário
  const userId = await createUser({ email, username, password });
  
  // Criar pacote draft
  const packageId = await createPackage({
    userId,
    ...packageData,
    status: 'draft'
  });
  
  // Enviar email com credenciais
  await sendWelcomeEmail(email, username, password, packageId);
  
  res.json({ success: true, packageId });
});
```
